Variables - Variable scope

  • no block scope but function scope!
  • with var => limited to its function block
  • without => define in the global object...

Variable scope 1/3 

        
var scope = 'global';
function checkScope(){
    var scope = 'local';
    return scope;
}
checkScope();

Variable scope 2/3

        
scope = 'global';
function checkScope(){
    scope = 'local';
    myScope = 'local';
    return [scope, myScope];
}
checkScope();
scope;
myScope;

Variable scope 3/3

        
var scope = 'global scope';
function checkScope(){
    var scope = 'local scope';
    function nested(){
        var scope = 'nested scope';
        return scope;
    }
    return nested();
}
checkScope();
scope;

  console.log(declaredButNotAssigned);

  console.log(declaredButNotAssigned);
}
example();
